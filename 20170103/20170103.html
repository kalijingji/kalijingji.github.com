常用jquery实用动画

点击文字变化图片

		$(function(){
		  $(".tab-item").mouseenter(function(){
			$(this).addClass("active").siblings().removeClass("active");//siblings()选择除了所选中的以外的所有兄弟元素
			$(".main").eq($(this).index()).addClass("selected").siblings().removeClass("selected");
		})
	    })
实例

图片轮播

		$(function(){
		 var idx = 0;
		 $(".wrap ol li").mouseenter(function(){
			idx+=1;
			$(this).addClass("active").siblings().removeClass("active");
			$(".wrap ul li").eq($(this).index()).css({"left":650,"z-index":idx}).animate({"left":0},300)
		})
	    })
实例

手风琴

		 $(function () {
            $(".wrap li").each(function (i,ele) {
                $(this).css("background-image","url(img/" + (i + 1) +".jpg)");
            }).mouseenter(function () {
                $(this).stop().animate({
                    "width":600
                },300).siblings().stop().animate({
                	"width":100 
                },300)
            }).mouseleave(function () {
                $("li").stop().animate({
                    "width":200
                },300)
            });
          })
实例

天猫小广告

		$(function(){
		$("#left li").mouseenter(function(){
			$(this).addClass()
			$("#center li").eq($(this).index()).show().siblings("li").hide();
		})
		$("#right li").mouseenter(function(){
			$("#center li").eq($(this).index()).show().siblings("li").hide();
		})
	    })
实例

图片跟随鼠标滚动

		 $(function(){
				$(window).scroll(function(){
					var tp = $(document).scrollTop();
					$("#left,#right").stop().animate({"top":tp},300)
				})
			})

透明动画

		  $(function(){
				$(".left").mouseenter(function(){
					$(".two").eq($(this).index()).show().stop().animate({"width":300},300);
					$(".nba").eq($(this).index()).show().stop().animate({"height":0},300).hide();
				}).mouseleave(function(){
					$(".two").eq($(this).index()).show().stop().animate({"width":0},300).hide();
					$(".nba").eq($(this).index()).show().stop().animate({"height":50},300);
				})
			})
