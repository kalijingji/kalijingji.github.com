			     var btn = document.getElementById("btn");
			     var t = 59;
			     btn.addEventListener("click",move,false);
			     var time;
                 function move(){
          	            clearInterval(time);
				        btn.innerHTML = t +"秒获取验证码";
				        btn.disabled = true;
                        time=window.setInterval(move1,1000);
			         }
                 function move1(){
          	
          	     btn.innerHTML = t +"秒获取验证码";
          	         if(t<=0){
          		       t = 59;  
          		          if(time){clearInterval(time)}//检测定时器是否存在
          	              btn.disabled = false;
          	              btn.innerHTML = "获取手机验证码";
          	               }
          	           t--;
                         }
		         </script>
