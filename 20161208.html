string对象

1.申明字符串的几种方式

			       var str="我是字符串";
					str="我是字符串";
					var str1=new string('我是字符串');
					str1=new string('我是字符串');

2.应用

			                var str1="我是字符串我"
					console.log(str1.length);//字符串的长度根据字符的位数来决定
					console.log(str1.indexOf("我"))//指定字符串首次出现位置
					console.log(str1.indexOf("我",1))  //或者写成
					console.log(str1.indexOf("我",str1.indexOf("我")+1))//指定字符串第二次出现位置

3.实例

		            var str = "[ti:The Fox][00:00.00] The Fox - Ylvis";
			        var m = str.split("]");
			        document.write("歌名:" + m[0].split(":")[1] + "<br/>");
			        document.write("时间:" + m[1].split("[")[1] + "<br/>");
			        document.write("歌手:" + m[2].split("-")[1]);

arr对象

1.申明数组的几种方式

			                        var arr = ["策划","技术总监","部门经理","项目组长"];
						var arr1 = [];
						var arr2 = new Array();
						var arr3 = new Array(1,3);

2.应用

a.pop

					console.log(arr.pop());//删除最后一个并返回

b.shift

					console.log(arr.shift());//删除第一个一个并返回
		
c.Array.is
		
		
					console.log(Array.isArray(arr));//判断是不是数组,是返回true
					

		
d.splice

		
			        console.log(arr.splice(2,1,"贾大师")+"    "+arr);//2代表			
                                从第三位起删除第一个数组内容,并且添加"贾大师"到删除的数组位置中       
                

		
e.slice

		
		            console.log(arr.slice(2,1)+"    "+arr);//拷贝一个数组
		            

		
f.forEach

		
			var arr = ["策划","技术总监","部门经理","项目组长"];
			arr.forEach(function(element,index,array){
			console.log("element:"+element);//依次返回数组
			console.log("index:"+index);//依次返回数组索引号
			console.log("array:"+array);//返回整个数组
		                        })
		

		
g.push

		
			        arr2.push("张三")//给数组里面添加一个元素
				console.log(arr2);//返回数组内容为张三
				console.log(arr2.push());//返回数组长度
