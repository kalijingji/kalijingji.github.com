jQuery添加css样式

          $(function(){
			$("#btn").click(function(){
            $("#img").removeAttr("src")
            console.log($("#box").text("red"));//相当于innerText;
            console.log($("#ipt").val("哈哈"));//相当于value.
            console.log($("#box").html("<span>red</span>"));//相当于innerHTML
             console.log($("#box").append("<p>哈哈</p>"));//添加原基础上添加元素或者文字
             console.log($("<p>哈哈</p>").appendTo("#box"));//添加原基础上添加元素或者文字
             console.log($("#box").prepend("<p>哈哈</p>"));//添加原基础上添加元素或者文字
             console.log($("<p>哈哈</p>").appendTo("#box"));//添加原基础上添加元素或者文字
             console.log($("#box").empty());//清空#box里面的所有子元素
             console.log($("#box").remove());//清空#box里面的所有子元素包括#box自身
             $("#box").clone().prependTo("#box");//克隆
             $("#box").addClass("Class");//添加css样式
             $("#box").removeClass("Class");//移除css样式
             $("#box").css("red");//直接添加css样式
             $("#box").css({"color":"red","background":"green"}); //设置多个属性用花括号扩起来.
            console.log($("p").innerHeight());//只算留白,不算边框
            console.log($("p").innerWidth());//只算留白,不算边框
            console.log($("p").outerHeight());//留白,边框都算
            console.log($("p").outerWidth());//留白,边框都算
			})
		})

jQuery动画

        	$("#btn").click(function(){
		$("#box").show(2000)//显示
	        $("#box").fadeIn(2000);//淡入
                $("#box").slideDown(2000);//
                $("#box").slideToggle(2000);//滑动出现消失
                $("#box").fadeToggle(2000);//淡入淡出
                $("#box").animate({ width: "90%",height: "100%", fontSize: "10em", borderWidth: 10},1000);//自定义动画
			})

ajax-jQuery用法

        	$(function(){
			$("#btn").click(function(){
				$.ajax({
					url:"txt.txt",//文件地址
					type:"GET",
					dataType:"text",
					data:"",
					success:function(msg){
						console.log("服务器返回成功:"+msg);
						$("#box").html(msg)
					},error:function(erroy){
						console.log("服务器返回失败:"+erroy);
					}
				})
			})
		})
